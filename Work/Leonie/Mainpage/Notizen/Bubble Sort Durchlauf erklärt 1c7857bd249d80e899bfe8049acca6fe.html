function bubbleSortV2(arr) {
    const n = arr.length;
    for (let i = 0; i < n - 1; i++) { // Äußere Schleife: Anzahl der Durchläufe
        printLn("Durchlauf " + (i + 1)); // Ausgabe des aktuellen Arrays
        for (let j = 0; j < n - i - 1; j++) { // Innere Schleife: Vergleiche benachbarte Elemente
            printLn("  Vergleich: " + arr[j] + " > " + arr[j + 1]); // Ausgabe der Vergleiche
            if (arr[j] > arr[j + 1]) { // Wenn das linke Element größer ist, tausche
                printLn("        Tausch: " + arr[j] + " Index:" + arr.indexOf(arr[j]) + " und " + arr[j + 1] + " Index:" + arr.indexOf(arr[j + 1])); // Ausgabe des Tauschs
                printLn("        Array vor Tausch:  " + arr); // Ausgabe des Arrays vor dem Tausch
                [arr[j], arr[j + 1]] = [arr[j + 1], arr[j]]; // Swap
                printLn("        Array nach Tausch: " + arr); // Ausgabe des Arrays nach dem Tausch
            }
            printHr(); // Ausgabe einer horizontalen Linie
        }
    }
    return arr; // Rückgabe des sortierten Arrays
}
